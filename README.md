    return((as.numeric(vals[1]) + as.numeric(vals[2])) / 2)
  } else {
    # Return the original value if there's no "/"
    return(as.numeric(x))

  case_when(
    month %in% c(12, 1, 2) ~ "Winter",
    month %in% c(3, 4, 5) ~ "Spring",
    month %in% c(6, 7, 8) ~ "Summer",
    month %in% c(9, 10, 11) ~ "Fall"
  )
